  // Runs Main.java
  >> ant
  >> sh run.sh
  >> ./run.sh

  // Compiles the source directory
  >> ant
  >> ant compile

  // Compiles the tests and runs them
  >> ant test

  // Deletes binaries
  >> ant clean

  // Generates jar file
  >> ant jar
